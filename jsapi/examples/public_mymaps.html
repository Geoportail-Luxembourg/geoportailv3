HTML to display the mymaps : 
<html>
  <head>
    <script src="//apiv3.geoportail.lu/apiv3loader.js"  type="text/javascript"></script>
  </head>
  <body>
    <div id="map1"></div>
    <script>
      var curMap = new lux.Map({
        target: 'map1',
        bgLayer: 'topo_bw_jpeg',
        zoom: 12,
        position: [76825, 75133]
      });
      curMap.addMyMapLayer({
        mapId: ''
      });
    </script>
  </body>
</html>
